# Astra PoC — Working Security Scanner with IBM Carbon Dashboard

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a demonstrable PoC: a Next.js 15 app using IBM Carbon Design System that accepts a GitHub repo URL, clones it, runs Trivy + Semgrep + Bearer + Gitleaks, normalizes findings, runs Ollama-based business logic inference, and displays all results in a Carbon-styled dashboard.

**Architecture:** Single Next.js 15 app with App Router, fully containerized. API routes execute scanner binaries via `child_process.exec` inside a Docker image that bakes in Trivy, Semgrep, Bearer, Gitleaks, Checkov, Bandit. SQLite for PoC storage. Ollama runs in a separate container reachable via `OLLAMA_HOST`. Dashboard uses `@carbon/react` components with IBM Plex Sans typography. Orchestrated with `docker compose up`.

**Tech Stack:** Next.js 15 (App Router), TypeScript, `@carbon/react`, `@carbon/styles`, IBM Plex Sans (Google Fonts), SQLite (`better-sqlite3`), Ollama (local), Trivy, Semgrep, Bearer, Gitleaks.

```typescript
    scanners_run TEXT NOT NULL DEFAULT '[]',

  CREATE TABLE IF NOT EXISTS findings (

    scanner TEXT NOT NULL,
    rule_id TEXT NOT NULL,
    title TEXT NOT NULL,
    description TEXT NOT NULL DEFAULT '',
    severity TEXT NOT NULL,
    category TEXT NOT NULL,
    file TEXT NOT NULL,

    code_snippet TEXT NOT NULL DEFAULT '',

    cwe TEXT NOT NULL DEFAULT '[]',
    owasp TEXT NOT NULL DEFAULT '[]',
    ai_explanation TEXT,
    ai_fix TEXT,
    exploit_score REAL,
    remediation TEXT NOT NULL DEFAULT '',
    raw TEXT NOT NULL DEFAULT '{}',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(fingerprint, scan_id)
  );

  CREATE TABLE IF NOT EXISTS biz_logic_rules (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    scan_id INTEGER NOT NULL REFERENCES scans(id),
    rule_text TEXT NOT NULL,
    confidence REAL NOT NULL,

    violation_description TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP

export function createScan(repoUrl: string, branch = 'main'): ScanMeta {
  const stmt = db.prepare(
    'INSERT INTO scans (repo_url, branch, status) VALUES (?, ?, ?) RETURNING *'
  );
  return stmt.get(repoUrl, branch, 'RUNNING') as ScanMeta;
}

export function completeScan(scanId: number, duration: number, scanners: string[]) {
  const stmt = db.prepare(
    'UPDATE scans SET status = ?, duration_seconds = ?, scanners_run = ? WHERE id = ?'
  );
  stmt.run('COMPLETED', duration, JSON.stringify(scanners), scanId);
}

export function insertFinding(scanId: number, f: Omit<UnifiedFinding, 'id' | 'createdAt'>) {
  const stmt = db.prepare(`
    INSERT INTO findings (fingerprint, scan_id, scanner, rule_id, title, description,
      severity, category, file, line_start, line_end, code_snippet, language,
      cwe, owasp, ai_explanation, ai_fix, exploit_score, remediation, raw)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
    ON CONFLICT(fingerprint, scan_id) DO UPDATE SET
      title = excluded.title,
      description = excluded.description,
      severity = excluded.severity
    RETURNING id
  `);
  return stmt.get(
    f.fingerprint, scanId, f.scanner, f.ruleId, f.title, f.description,
    f.severity, f.category, f.file, f.lineStart, f.lineEnd, f.codeSnippet, f.language,
    JSON.stringify(f.cwe), JSON.stringify(f.owasp), f.aiExplanation, f.aiFix, f.exploitScore,
    f.remediation, f.raw
  ) as { id: number };
}

export function insertBizRule(scanId: number, rule: Omit<BusinessLogicRule, 'id' | 'createdAt'>) {
  const stmt = db.prepare(`
    INSERT INTO biz_logic_rules (scan_id, rule_text, confidence, evidence_files, status, violation_description)
    VALUES (?, ?, ?, ?, ?, ?)
  `);
  stmt.run(scanId, rule.ruleText, rule.confidence, JSON.stringify(rule.evidenceFiles), rule.status, rule.violationDescription);

export function listFindings(scanId: number): UnifiedFinding[] {
```

```typescript
import { execSync } from 'child_process';
import fs from 'fs';
import path from 'path';

export interface ScannerResult {
  scanner: string;
  findings: RawFinding[];
  durationMs: number;
  error?: string;
}

export interface RawFinding {
  ruleId: string;
  title: string;
  description: string;
  severity: string;
  file: string;
  line: number;
  code?: string;
  cwe?: string[];
}

const REPO_DIR = '/tmp/astra-repo';

export function cloneRepo(url: string): string {
  if (fs.existsSync(REPO_DIR)) {
    fs.rmSync(REPO_DIR, { recursive: true });
  }
  execSync(`git clone --depth 1 ${url} ${REPO_DIR}`, { stdio: 'pipe', timeout: 60000 });
  return REPO_DIR;
}

export function runTrivy(repoDir: string): ScannerResult {
  const start = Date.now();
  try {
    execSync(`trivy fs --scanners vuln,misconfig,secret --format json -o /tmp/trivy.json ${repoDir}`, {
      stdio: 'pipe', timeout: 120000,
    });
    const raw = JSON.parse(fs.readFileSync('/tmp/trivy.json', 'utf8'));
    const findings: RawFinding[] = [];
    for (const result of raw.Results ?? []) {
      for (const vuln of result.Vulnerabilities ?? []) {
        findings.push({
          ruleId: vuln.VulnerabilityID ?? 'unknown',
          title: vuln.Title ?? vuln.VulnerabilityID ?? 'Unknown vulnerability',
          description: vuln.Description ?? '',
          severity: vuln.Severity ?? 'UNKNOWN',
          file: result.Target ?? 'unknown',
          line: vuln.Line ?? 0,
          cwe: vuln.CweIDs ?? [],
        });
      }
      for (const misconfig of result.Misconfigurations ?? []) {
        findings.push({
          ruleId: misconfig.ID ?? 'unknown',
          title: misconfig.Title ?? 'Misconfiguration',
          description: misconfig.Description ?? '',
          severity: misconfig.Severity ?? 'UNKNOWN',
          file: result.Target ?? 'unknown',
          line: misconfig.Line ?? 0,
        });
      }
      for (const secret of result.Secrets ?? []) {
        findings.push({
          ruleId: secret.RuleID ?? 'secret',
          title: `Secret: ${secret.Title ?? 'detected'}`,
          description: secret.Match ?? '',
          severity: secret.Severity ?? 'HIGH',
          file: result.Target ?? 'unknown',
          line: secret.StartLine ?? 0,
        });
      }
    }
    return { scanner: 'trivy', findings, durationMs: Date.now() - start };
  } catch (e: any) {
    return { scanner: 'trivy', findings: [], durationMs: Date.now() - start, error: e.message };
  }
}

export function runSemgrep(repoDir: string): ScannerResult {
  const start = Date.now();
  try {
    execSync(`semgrep --config=auto --json -o /tmp/semgrep.json ${repoDir}`, {
      stdio: 'pipe', timeout: 120000,
    });
    const raw = JSON.parse(fs.readFileSync('/tmp/semgrep.json', 'utf8'));
    const findings: RawFinding[] = [];
    for (const result of raw.results ?? []) {
      findings.push({
        ruleId: result.check_id ?? 'unknown',
        title: result.extra?.message ?? result.check_id ?? 'Finding',
        description: result.extra?.message ?? '',
        severity: result.extra?.severity ?? 'WARNING',
        file: result.path ?? 'unknown',
        line: result.start?.line ?? 0,
        code: result.extra?.lines ?? '',
        cwe: result.extra?.metadata?.cwe ?? [],
      });
    }
    return { scanner: 'semgrep', findings, durationMs: Date.now() - start };
  } catch (e: any) {
    return { scanner: 'semgrep', findings: [], durationMs: Date.now() - start, error: e.message };
  }
}

export function runGitleaks(repoDir: string): ScannerResult {
  const start = Date.now();
  try {
    execSync(`gitleaks detect --source ${repoDir} --report-format json --report-path /tmp/gitleaks.json -v`, {
      stdio: 'pipe', timeout: 60000,
    });
    const raw = JSON.parse(fs.readFileSync('/tmp/gitleaks.json', 'utf8'));
    const findings: RawFinding[] = [];
    for (const finding of raw ?? []) {
      findings.push({
        ruleId: finding.RuleID ?? 'secret',
        title: `Secret leak: ${finding.Description ?? finding.RuleID}`,
        description: finding.Match ?? '',
        severity: 'HIGH',
        file: finding.File ?? 'unknown',
        line: finding.StartLine ?? 0,
      });
    }
    return { scanner: 'gitleaks', findings, durationMs: Date.now() - start };
  } catch (e: any) {
    // gitleaks exits 1 when findings found — that's success for us
    try {
      const raw = JSON.parse(fs.readFileSync('/tmp/gitleaks.json', 'utf8'));
      const findings: RawFinding[] = [];
      for (const finding of raw ?? []) {
        findings.push({
          ruleId: finding.RuleID ?? 'secret',
          title: `Secret leak: ${finding.Description ?? finding.RuleID}`,
          description: finding.Match ?? '',
          severity: 'HIGH',
          file: finding.File ?? 'unknown',
          line: finding.StartLine ?? 0,
        });
      }
      return { scanner: 'gitleaks', findings, durationMs: Date.now() - start };
    } catch {
      return { scanner: 'gitleaks', findings: [], durationMs: Date.now() - start, error: e.message };
    }
  }
}

export function runBearer(repoDir: string): ScannerResult {
  const start = Date.now();
  try {
    execSync(`bearer scan ${repoDir} --format json --output /tmp/bearer.json`, {
      stdio: 'pipe', timeout: 120000,
    });
    const raw = JSON.parse(fs.readFileSync('/tmp/bearer.json', 'utf8'));
    const findings: RawFinding[] = [];
    for (const finding of raw.data ?? []) {
      findings.push({
        ruleId: finding.rule_id ?? 'unknown',
        title: finding.title ?? 'Data flow finding',
        description: finding.description ?? '',
        severity: finding.severity ?? 'MEDIUM',
        file: finding.file ?? 'unknown',
        line: finding.line_number ?? 0,
      });
    }
    return { scanner: 'bearer', findings, durationMs: Date.now() - start };
  } catch (e: any) {
    return { scanner: 'bearer', findings: [], durationMs: Date.now() - start, error: e.message };
  }
}

export function runAllScanners(repoDir: string): ScannerResult[] {
  return [
    runTrivy(repoDir),
    runSemgrep(repoDir),
    runGitleaks(repoDir),
    runBearer(repoDir),
  ];
}
```

```typescript
import { createHash } from 'crypto';
import type { RawFinding, ScannerResult } from './scanners';
import type { UnifiedFinding, Severity, Category } from './types';

function mapSeverity(s: string): Severity {
  const upper = s.toUpperCase();
  if (['CRITICAL', 'CRIT'].includes(upper)) return 'CRITICAL';
  if (['HIGH', 'ERROR'].includes(upper)) return 'HIGH';
  if (['MEDIUM', 'MODERATE', 'WARN', 'WARNING'].includes(upper)) return 'MEDIUM';
  if (['LOW', 'INFO', 'INFORMATIONAL'].includes(upper)) return 'LOW';
  return 'INFO';
}

function mapCategory(scanner: string): Category {
  switch (scanner) {
    case 'trivy': return 'SCA';
    case 'semgrep': return 'SAST';
    case 'gitleaks': return 'SECRETS';
    case 'bearer': return 'DATA_FLOW';
    default: return 'SAST';
  }
}

function fingerprint(scanner: string, ruleId: string, file: string, line: number, code?: string): string {
  const hash = createHash('sha256');
  hash.update(`${scanner}:${ruleId}:${file}:${line}:${code ?? ''}`);
  return hash.digest('hex').slice(0, 32);
}

export function normalize(result: ScannerResult): UnifiedFinding[] {
  return result.findings.map((f: RawFinding) => ({
    id: 0,
    fingerprint: fingerprint(result.scanner, f.ruleId, f.file, f.line, f.code),
    scanner: result.scanner,
    ruleId: f.ruleId,
    title: f.title,
    description: f.description,
    severity: mapSeverity(f.severity),
    category: mapCategory(result.scanner),
    file: f.file,
    lineStart: f.line,
    lineEnd: f.line,
    codeSnippet: f.code ?? '',
    language: '',
    cwe: f.cwe ?? [],
    owasp: [],
    aiExplanation: null,
    aiFix: null,
    exploitScore: null,
    remediation: '',
    raw: JSON.stringify(f),
    createdAt: '',
  }));
}
```

- [ ] **Step 1: Write Ollama client**

```typescript
import type { UnifiedFinding, BusinessLogicRule } from './types';

const OLLAMA_URL = process.env.OLLAMA_URL ?? 'http://localhost:11434';
const MODEL = process.env.OLLAMA_MODEL ?? 'codellama:13b';

async function ollamaChat(system: string, prompt: string): Promise<string> {
  const res = await fetch(`${OLLAMA_URL}/api/generate`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      model: MODEL,
      system,
      prompt,
      stream: false,
      options: { temperature: 0.2, num_predict: 512 },
    }),
  });
  if (!res.ok) throw new Error(`Ollama error: ${res.status}`);
  const json = await res.json();
  return json.response ?? '';
}

export async function enrichFinding(f: UnifiedFinding): Promise<Partial<UnifiedFinding>> {
  const system = 'You are an experienced application security engineer. Be concise.';
  const prompt = `
Finding: ${f.title} (${f.severity}) in ${f.file}:${f.lineStart}
Category: ${f.category}
Description: ${f.description}
Code snippet: ${f.codeSnippet.slice(0, 500)}

Provide:
1. One-sentence plain-English risk explanation
2. A concrete code fix snippet in the same language
3. Exploitability score 0-10 (just the number)

Format as JSON: {"explanation": "...", "fix": "...", "score": N}
`;
  try {
    const raw = await ollamaChat(system, prompt);
    // Extract JSON from markdown code blocks or raw text
    const match = raw.match(/\{[\s\S]*?\}/);
    if (!match) return {};
    const parsed = JSON.parse(match[0]);
    return {
      aiExplanation: parsed.explanation ?? null,
      aiFix: parsed.fix ?? null,
      exploitScore: typeof parsed.score === 'number' ? parsed.score : null,
    };
  } catch {
    return {};
  }
}

export async function inferBusinessLogic(repoDir: string, files: string[]): Promise<BusinessLogicRule[]> {
  const system = 'You are an expert at inferring business rules from application source code. Be concise.';
  // Read up to 5 entry-point files (controllers, handlers)
  const sampleFiles = files.slice(0, 5);
  const code = sampleFiles.map(f => `--- ${f} ---\n` + readFileSafe(`${repoDir}/${f}`).slice(0, 2000)).join('\n\n');

  const prompt = `
Given these source files, infer the most important business logic rules that govern how this application should behave.

Code:
${code}

List 3-5 rules. For each: state the rule, your confidence (0.0-1.0), and which files support it.

Format as JSON array:
[{"rule": "...", "confidence": 0.95, "evidence": ["file1.ts", "file2.ts"]}]
`;
  try {
    const raw = await ollamaChat(system, prompt);
    const match = raw.match(/\[[\s\S]*?\]/);
    if (!match) return [];
    const parsed = JSON.parse(match[0]);
    return parsed.map((r: any) => ({
      id: 0,
      ruleText: r.rule ?? 'Unknown rule',
      confidence: typeof r.confidence === 'number' ? r.confidence : 0.5,
      evidenceFiles: Array.isArray(r.evidence) ? r.evidence : [],
      status: 'CANDIDATE',
      violationDescription: null,
      createdAt: '',
    }));
  } catch {
    return [];
  }
}

function readFileSafe(path: string): string {
  try {
    return require('fs').readFileSync(path, 'utf8');
  } catch {
    return '';
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { cloneRepo, runAllScanners } from '@/lib/scanners';
import { normalize } from '@/lib/normalize';
import { enrichFinding, inferBusinessLogic } from '@/lib/ai';
import { createScan, completeScan, insertFinding, insertBizRule } from '@/lib/db';
import { globSync } from 'glob';

export async function POST(request: NextRequest) {
  const body = await request.json();
  const repoUrl: string = body.repoUrl;
  if (!repoUrl || !repoUrl.startsWith('https://github.com/')) {
    return NextResponse.json({ error: 'Valid GitHub URL required' }, { status: 400 });
  }

  const startTime = Date.now();
  const scan = createScan(repoUrl);

  try {
    const repoDir = cloneRepo(repoUrl);
    const scannerResults = runAllScanners(repoDir);
    const scannersRun = scannerResults.map(r => r.scanner);

    for (const result of scannerResults) {
      const findings = normalize(result);
      for (const f of findings) {
        const enriched = await enrichFinding(f);
        insertFinding(scan.id, { ...f, ...enriched });
      }
    }

    // Business logic inference
    const codeFiles = globSync('**/*.{js,ts,jsx,tsx,go,py,java,rb}', { cwd: repoDir, absolute: false });
    const rules = await inferBusinessLogic(repoDir, codeFiles.slice(0, 20));
    for (const rule of rules) {
      insertBizRule(scan.id, rule);
    }

    completeScan(scan.id, Math.round((Date.now() - startTime) / 1000), scannersRun);
    return NextResponse.json({ scanId: scan.id, findingsCount: scannerResults.reduce((sum, r) => sum + r.findings.length, 0), rulesCount: rules.length });
  } catch (e: any) {
    return NextResponse.json({ error: e.message, scanId: scan.id }, { status: 500 });
  }
}
```
